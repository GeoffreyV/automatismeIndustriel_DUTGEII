\newcommand\evaluationProf[1][1]{
\def\max{#1}
  \textit{(#1 pt(s))}
  \bareme{default.GRAD=0, formula=\max*GRAD/5, MAX=\max}
  \AMCOpen{lineup=true}%annotate=FALSE,framerule=0pt,dots=FALSE,lines=0}
  {%
  \wrongchoice[0]{0}\scoring{0}
  \wrongchoice[1]{1}\scoring{set.GRAD=1}
  \wrongchoice[2]{2}\scoring{set.GRAD=2}
  \wrongchoice[3]{3}\scoring{set.GRAD=3}
  \wrongchoice[4]{4}\scoring{set.GRAD=4}
  \correctchoice[5]{5}\scoring{set.GRAD=5}}%
}

\newcommand\checkProf[1][1]{
\def\max{#1}
  \textit{(#1 pt(s))}
  \bareme{default.GRAD=0, formula=\max*GRAD, MAX=\max}
  \AMCOpen{lineup=true}%annotate=FALSE,framerule=0pt,dots=FALSE,lines=0}
  {
  \wrongchoice[0]{0}\scoring{0}
  \correctchoice[1]{1}\scoring{set.GRAD=1}}
}

\newcommand{\checkProfQuestion}[3]{
\begin{question}{#1}
  #2
  \checkProf[#3]
\end{question}
}


\newcommand{\notationProf}[1]{

  \textit{(#1 pt(s))}
  \bareme{default.UN=0, default.DEUX=0, default.QUATRE=0, default.HUIT=0, formula=UN + DEUX +QUATRE +HUIT, MAX=\max}
  \AMCOpen{lineup=true}%annotate=FALSE,framerule=0pt,dots=FALSE,lines=0}
  {
  \wrongchoice[0]{0}\scoring{0}
  \correctchoice[1]{1}\scoring{set.UN=1}
  \correctchoice[2]{2}\scoring{set.DEUX=2}
  \correctchoice[4]{4}\scoring{set.QUATRE=4}
  \correctchoice[8]{8}\scoring{set.HUIT=8}
  }
}
